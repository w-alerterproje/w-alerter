        function isInside(point, vs) {

            var x = point[0], y = point[1], inside = false;

            for (var i = 0, j = vs.length - 1; i < vs.length; j = i++) {

                var xi = vs[i][0], yi = vs[i][1], xj = vs[j][0], yj = vs[j][1];

                var intersect = ((yi > y) != (yj > y)) && (x < (xj - xi) * (y - yi) / (yj - yi) + xi);

                if (intersect) inside = !inside;

            }

            return inside;

        }


        function isObstacle(lat, lng) {

            if (lat < -82.5) return true;

            return isInside([lat, lng], antarcticaPolygon);

        }

        function getSafetyCost(lat, lon) {

            if (isObstacle(lat, lon)) return 1000000;

            const month = document.getElementById('monthPicker').value;

            const data = environmentalDatabase[month] || [];

            let riskScore = 1;

            data.forEach(p => {

                let d = Math.sqrt(Math.pow(p.lat - lat, 2) + Math.pow(p.lon - lon, 2));

                if (d < 6.0 && p.chl >= 0.15) {

                    riskScore += Math.pow((6.0 - d) * (p.chl * 10), 2);

                }

            });

            return riskScore;

        }


        function startBioNavigation() {

            if (waypoints.length < 2) { showNotif("HATA: En az iki waypoint gereklidir!"); return; }

            document.getElementById('status-display').innerText = "Rota Sabırla Hesaplanıyor...";

            

            let fullPath = [];

            for (let i = 0; i < waypoints.length - 1; i++) {

                let segment = findHydraPath(waypoints[i], waypoints[i+1]);

                if (!segment) {

                    showNotif("KRİTİK HATA: Engeller aşılamadı! Lütfen noktayı deniz üzerine taşıyın.");

        }


        function findHydraPath(start, end) {

            const step = 0.45;

            let openSet = [{ pos: start, path: [start], g: 0, h: Math.sqrt(distSq(start, end)) * 1.1 }];

            let closedSet = new Set();

            const toKey = (p) => `${Math.round(p[0]*20)},${Math.round(p[1]*20)}`;

            let iters = 0;

            const maxIters = 30000;

            while (openSet.length > 0 && iters < maxIters) {

                iters++;

                openSet.sort((a, b) => (a.g + a.h) - (b.g + b.h));

                let curr = openSet.shift();

                if (distSq(curr.pos, end) < step * step * 3.5) { curr.path.push(end); return curr.path; }

                let key = toKey(curr.pos);

                if (closedSet.has(key)) continue;

                closedSet.add(key);

                const neighbors = [[step,0],[-step,0],[0,step],[0,-step],[step,step],[step,-step],[-step,step],[-step,-step]];

                for (let [dLat, dLon] of neighbors) {

                    let nPos = [curr.pos[0] + dLat, curr.pos[1] + dLon];

                    if (isObstacle(nPos[0], nPos[1])) continue;

                    let cost = getSafetyCost(nPos[0], nPos[1]);

                    let g = curr.g + step * cost;

                    let h = Math.sqrt(distSq(nPos, end)) * 1.1;

                    openSet.push({ pos: nPos, path: [...curr.path, nPos], g: g, h: h });

                }

            }

            return null;

        }


        function distSq(a, b) { return Math.pow(a[0]-b[0], 2) + Math.pow(a[1]-b[1], 2); }

        function updateSystem() {

            const chl = parseFloat(document.getElementById('in-chl').value) || 0.25;

            const sst = parseFloat(document.getElementById('in-sst').value) || 2.5;

            const weight = parseFloat(document.getElementById('in-weight').value) || 12000;

            const fuel = document.getElementById('in-fuel').value;

            const time = document.getElementById('in-time').value;

            const dist = parseFloat(document.getElementById('out-dist').innerText) || 0;

            

            // 1. Lojistik Regresyon Formülü (Mavi Balina Olasılığı - P)

            const z = -3.2 + 10.5 * chl + 0.25 * sst;

            let prob = 1 / (1 + Math.exp(-z));

            prob = prob * 100; // Yüzdeye çevir


            // Gece Seyri Faktörü (Riski %20 artırır)

            if (time === 'Gece') {

                prob = prob * 1.20;

            }

            prob = Math.min(Math.max(prob, 5.0), 98.8); // Sınırla


            // 2. Önerilen Hız Limitleri (Rapordaki Bantlara Göre)

            let speed = 19.8; // Güvenli Hızlı (Alt Sınır)

            if (prob > 60) {

                speed = 8.2;  // Çok Yavaş Seyir

            } else if (prob > 30) {

                speed = 12.5; // Yavaş Seyir

            }


            // 3. Güç-Hız Kübik Tasarruf Formülü (Ref: 24 Kts)

            const savings = (1 - Math.pow(speed / 24, 3)) * 100;


            // 4. Su Altı Gürültü Seviyesi (NL, dB)

            const logV = Math.log10(speed / 12);

            const logW = Math.log10(weight / 10000);

            const noise = 120 + 20 * logV + 15 * logW;


            // 5. Karbon Emisyon Modeli (Cübik Tüketim & Faktörler)

            const fc = 0.001 * Math.pow(speed / 12, 3) * Math.pow(weight / 10000, 2/3); // ton/Nm

            const totalFuelKg = (dist * fc) * 1000;

            

            let ef = 3.179; // Orijinal Dizel CO2 Emisyon Katsayısı

            if (fuel === "LNG") ef = 2.755; // LNG Katsayısı

            if (fuel === "Hidrojen") ef = 0.0; // Sıfır Karbon


            const co2 = totalFuelKg * ef;


            // Arayüz Çıktılarını Güncelle

            document.getElementById('out-prob').innerText = `%${prob.toFixed(1)}`;

            document.getElementById('out-speed').innerText = `${speed.toFixed(1)} Kts`;

            document.getElementById('out-savings').innerText = `%${savings.toFixed(1)}`;

            document.getElementById('out-noise').innerText = `${noise.toFixed(1)} dB`;

            document.getElementById('out-co2').innerText = `${co2.toFixed(1)} kg`;

        }
